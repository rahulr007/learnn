Word count
create or replace procedure wordcount(str varchar2)
is
t varchar2(20);
cn number:=0;
begin
  for i in 1..length(str) loop
    t:=substr(str,i,1);
    if(t=' ') then
      cn:=cn+1;
    end if;
  end loop;
dbms_output.put_line(cn+1);
end;

Sum of even numbers/Sum of odd numbers
 create or replace procedure sumofodd(num1 in number,num2 in number)
is
sumeven number:=0;
sumodd number:=0;
begin
  for i in num1..num2 loop
     if(mod(i,2)=0) then
        sumeven:=sumeven+i;
     else
        sumodd:=sumodd+i;
     end if;
  end loop;
dbms_output.put_line('sum of even: '||sumeven);
dbms_output.put_line('sum of odd: '||sumodd);
end;
/



Number Palindrome 
create or replace procedure palindrome(a in number) is
num number;
sum1 number:=0;
temp number;
begin
                temp:=a;
                while(temp>0)
                loop
                                num:=mod(temp,10);
                                sum1:=sum1*10+num;
                                temp:=trunc(temp/10);
                end loop;
                if(sum1 = a) then
                                dbms_output.put_line('palindrome');
                else
                                dbms_output.put_line('not palindrome');
                end if;
end;
/

is

Reverse a String

create or replace function try_rev(val_1 in varchar) return varchar is
l VARCHAR2(20);  
    t VARCHAR2(10);  
   begin
    FOR i IN REVERSE 1..Length(val_1) 
         LOOP  
        l := Substr(val_1, i, 1);  
        t := t || l;  
    END LOOP;  
    return (t);
    exception
    when others then
    return 
         null;
    end;

 l VARCHAR2(20);  
    t VARCHAR2(10);  
BEGIN 
    FOR i IN REVERSE 1..Length(a) 
LOOP  
        l := Substr(a, i, 1);  
  
       
        t := t || l;

    IF t = a THEN 
      dbms_output.Put_line(t  ||' is palindrome');
